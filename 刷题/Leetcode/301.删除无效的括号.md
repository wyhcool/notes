如果我们每次只删除一个括号，然后观察被删除一个括号后是否合法，如果已经合法了，我们就不用继续删除了啊。因此我们并不需要将遍历进行到底，而是层层深入，一旦达到需求，就不再深入了。
```javascript
/**
 * @param {string} s
 * @return {string[]}
 */
var removeInvalidParentheses = function (s) {
  var len = s.length
  var visited = {}
  var nexted = {}
  var queue = [s]
  var ans = []
  var isFound = false

  var isValid = function (s) {
    var cnt = 0
    for (var i = 0; i < s.length; i++) {
        if (s[i] === '(') {
            cnt++
        } else if (s[i] === ')') {
            cnt--
        }
        if (cnt < 0) {
            return false
        }
    }
    return cnt === 0
  }

  while (len > 0) {
    if (isFound) {
      break
    }
    var nextQueue = []
    for (var k = 0; k < queue.length; k++) {
      var str = queue[k]
      if (str in visited) {
        continue
      }
      visited[str] = 1
      if (isValid(str)) {
        isFound = true
        ans.push(str)
      }
      if (!isFound) {
        // 如果我们每次只删除一个括号，然后观察被删除一个括号后是否合法，如果已经合法了，我们就不用继续删除了。
        // 因此我们并不需要将遍历进行到底，而是层层深入，一旦达到需求，就不再深入了。
        for (var i = 0; i < str.length; i++) {
          var nextStr = str.slice(0, i) + str.slice(i + 1)
          if (nextStr in nexted) {
            continue
          }
          nexted[nextStr] = 1
          nextQueue.push(nextStr)
        }
      }
    }
    queue = nextQueue
    len--
  }
  if (ans.length === 0) {
    return ['']
  }
  return ans
}
```
