画图分析与代码实现：
```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} n
 * @return {ListNode}
 */
var removeNthFromEnd = function(head, n) {
    var curr = head
    var mid_index = 0
    var count = 1
    while (curr) {
        var next = curr.next
        if (!next) {
            // 奇数个
            count = 2 * mid_index + 1
            break
        }
        var next2 = next.next
        if (!next2) {
            // 偶数个
            count = 2 * mid_index + 2
            break
        }
        curr = next2
        mid_index++
    }
    var del_index = count - n
    if (del_index === 0) {
        return head.next
    }
    var prev = head
    var curr = head
    for (var i = 0; i < del_index; i++) {
        prev = curr
        curr = curr.next
    }
    prev.next = curr.next
    return head
};
```
