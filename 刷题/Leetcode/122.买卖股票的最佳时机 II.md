遍历数组寻找最大的上涨空间。

```javascript
/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (prices) {
  var n = prices.length
  if (n <= 1) {
    return 0
  }
  var buy = prices[0]
  var profit = 0
  for (var i = 1; i < n; i++) {
    if (prices[i] > buy) {
      profit += prices[i] - buy
    }
    buy = prices[i]
  }
  return profit
}
```
