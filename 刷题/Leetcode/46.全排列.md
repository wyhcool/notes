/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var swap = function(array, i, j) {
    var temp = array[i]
    array[i] = array[j]
    array[j] = temp
}
var gen = function(nums, start, ans) {
    var n = nums.length
    if (start >= n) {
        var res = []
        for (var i = 0; i < n; i++) {
            res.push(nums[i])
        }
        ans.push(res)
        return
    }
    for (var i = start; i < n; i++) {
        swap(nums, i, start)
        gen(nums, start+1, ans)
        swap(nums, i, start)
    }
}
var permute = function(nums) {
    var ans = []

    gen(nums, 0, ans)

    return ans
};
